109 Nohm.prototype.init = function (name, options) {
110   if ( ! name )
111     this.logError('When creating a new model you have to provide a name!');
112     
113   if (typeof(options.client) === 'undefined' && Nohm.client === null)
114     return Nohm.logError('Did not find a viable redis client in Nohm or the model: '+name) && false;
115     
116   if (typeof(options.client) !== 'undefined') {
117     this.client = options.client;
118   }
119     
120   this.modelName = name;
121   
122   this.properties = {};
123   this.errors = {};
124   
125   // initialize the properties
126   if (options.hasOwnProperty('properties')) {
127     for (var p in options.properties) {
128       if (options.properties.hasOwnProperty(p)) {
129         this.properties[p] = h.$extend(true, {}, options.properties[p]); // deep copy
130         this.property(p, options.properties[p].value || 0); // this ensures typecasing/behaviours
131         this.__resetProp(p);
132         this.errors[p] = [];
133     
134         if ( Nohm.meta && ! Nohm.meta_saved_models.hasOwnProperty(this.modelName)) {
135           // try saving the meta data of this model
136           var metaargs = [Nohm.prefix.meta + this.modelName, p, JSON.stringify(this.properties[p])];
137           this.getClient().hmset(metaargs);
138         }
139       }
140     }
141   }
142 
143   this.relationChanges = [];
144 
145   this.id = null;
146   this.__inDB = false;
147   this.__loaded = false;
148 };
