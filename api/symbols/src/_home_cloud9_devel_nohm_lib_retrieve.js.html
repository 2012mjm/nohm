  9 /**
 10  * Check if a given id exists in the DB.
 11  */
 12 exports.exists = function (id, callback) {
 13   this.getClient().sismember(Nohm.prefix.idsets+this.modelName, id, function (err, found) {
 14     Nohm.logError(err);
 15     callback(!!found);
 16   });
 17 };

 19 /**
 20  * Retrieves the hash data by id and puts it into the properties.
 21  */
 22 exports.load = function (id, callback) {
 23   var self = this;
 24   id = parseInt(id, 10);
 25   if (isNaN(id) || id < 1) {
 26     Nohm.logError('Trying to pass load() a wrong kind of id. Needs to be a number over 0. (still trying to load it though)');
 27   }
 28   this.getClient().hgetall(this.getHashKey(id), function (err, values) {
 29     var p, value,
 30         keys = Object.keys(values);
 31     if (err) {
 32       Nohm.logError('loading a hash produced an error: ' + err);
 33     }
 34     if (Array.isArray(keys) && keys.length > 0) {
 35       for (p in values) {
 36         if (values.hasOwnProperty(p)) {
 37           value = values[p] !== null ? values[p].toString() : null;
 38           if (self.properties[p].load_pure) {
 39             self.properties[p].value = value;
 40           } else {
 41             self.p(p, value);
 42           }
 43           self.__resetProp(p);
 44         }
 45       }
 46       self.id = id;
 47       self.__inDB = true;
 48       self.__loaded = true;
 49     } else if (!err) {
 50       err = 'not found';
 51     }
 52     if (typeof(callback) === 'function') {
 53       callback(err);
 54     }
 55   });
 56 }

 59 /**
 60  * Finds ids of objects by search arguments
 61  */
 62 exports.find = function find(searches, callback) {
 63   var self = this, sets = [], zsetKeys = [], s, prop,
 64   returnFunction = function (err, values) {
 65     var found = [];
 66     Nohm.logError(err);
 67     if (Array.isArray(values) && values.length > 0) {
 68       values = values.forEach(function (val, i) {
 69         if (val) {
 70           var id = parseInt(val.toString(), 10);
 71           if (id > 0) {
 72             found.push(id);
 73           }
 74         }
 75       });
 76     }
 77     callback(err, found);
 78   },
 79   getSets = function (callback) {
 80     self.getClient().sinter(sets, callback);
 81   },
 82   getSingleZSet = function (zSet, callback) {
 83     var rangeCallback = function (err, values) {
 84       if (err) {
 85         callback(err);
 86       } else {
 87         values.forEach(function (val, key) {
 88           values[key] = parseInt(val, 10);
 89         });
 90         callback(null, values);
 91       }
 92     };
 93     options = zSet.options;
 94     if ( ! options.min && options.min !== 0)
 95       options.min = '-inf';
 96     if ( ! options.max && options.max !== 0)
 97       options.max = '+inf';
 98     if ( ! options.offset && options.offset !== 0)
 99       options.offset = '+inf';
100     if (options.limit) {
101       self.getClient().zrangebyscore(zSet.key, options.min, options.max,
102                       'LIMIT', options.offset, options.limit,
103                       rangeCallback);
104     } else {
105       self.getClient().zrangebyscore(zSet.key, options.min, options.max,
106                       rangeCallback);
107     }
108   },
109   getZSets = function (callback) {
110     async.map(zsetKeys, getSingleZSet, function done (err, arr) {
111       var test = h.idIntersection.apply(null, arr);
112       callback(null, test.sort());
113     });
114   };
115   
116   
117   if (typeof searches === 'function') {
118     callback = searches;
119     searches = {};
120   }
121   for (s in searches) {
122     if (searches.hasOwnProperty(s) && this.properties.hasOwnProperty(s)) {
123       prop = this.properties[s];
124       if (prop.unique) {
125         return this.getClient().mget(Nohm.prefix.unique + self.modelName + ':' + s +
126                          ':' + searches[s],
127                          returnFunction);
128       }
129       var isNum = !isNaN(parseInt(searches[s], 10));
130       if (prop.index && ( ! prop.__numericIndex || isNum) ) {
131         sets.push(Nohm.prefix.index + self.modelName + ':' + s + ':' + searches[s]);
132       } else if (prop.__numericIndex) {
133         zsetKeys.push({
134           key: Nohm.prefix.scoredindex + self.modelName + ':' + s,
135           options: searches[s]
136         });
137       }
138     }
139   }
140   if (sets.length === 0 && zsetKeys.length === 0) {
141     this.getClient().smembers(Nohm.prefix.idsets + this.modelName, function (err, ids) {
142       if (Array.isArray(ids)) {
143         ids = ids.map(function (val) {
144           return parseInt(val.toString(), 10);
145         });
146       }
147       returnFunction(err, ids);
148     });
149   } else if (zsetKeys.length === 0) {
150     getSets(returnFunction);
151   } else if (sets.length === 0) {
152     getZSets(returnFunction);
153   } else {
154     getSets(function (err, setids) {
155       if (Array.isArray(setids)) {
156         setids = setids.map(function (val) {
157           return parseInt(val.toString(), 10);
158         });
159       }
160       getZSets(function (err2, zsetids) {
161         if (err2) {
162           err = [err, err2];
163         }
164         returnFunction(err, h.idIntersection(setids, zsetids).sort());
165       });
166     });
167   }
168 }
