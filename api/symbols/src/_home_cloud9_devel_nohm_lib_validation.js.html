  9 /**
 10  *  Check if one or all propert(y/ies) are valid.
 11  * @see Validators
 12  */
 13 exports.valid = function valid(key, setDirectly) {
 14   var validbool = true,
 15   p,
 16   self = this,
 17   noKeySpecified = !key || typeof(key) === 'function',
 18   callback = h.getCallback(arguments);
 19   for (p in this.properties) {
 20     if (noKeySpecified || key === p) {
 21       if (!this.__validateProperty(p)) {
 22         validbool = false;
 23       }
 24     }
 25   }
 26   if (typeof callback === 'function') {
 27     // if the object isn't valid to begin with, we don't set the uniques and just check them
 28     setDirectly = validbool ? !!setDirectly : false;
 29     
 30     // if there is no valid callback, there is no good reason to call an async function.
 31     // TODO: determine whether this is true. if setDirectly is true then calling it
 32     // without a callback could have some use. (like blocking uniques :/)
 33     this.__checkUniques(setDirectly, function (success) {
 34 
 35       if (!key && !success) {
 36         validbool = false;
 37       } else if (key && !success) {
 38         validbool = !!self.errors[key];
 39       }
 40       callback(validbool);
 41     });
 42   }
 43   return validbool;
 44 };

 46 /**
 47  *  Check whether all properties marked as unique have a value that is not set in the database yet.
 48  *  If setDirectly is set, it will occupy the unique in the db directly.
 49  *  Use setDirectly if you're saving the object to prevent race-conditions.
 50  */
 51 exports.__checkUniques = function __checkUniques(setDirectly, saveCallback) {
 52   setDirectly = setDirectly || false;
 53   var functions = {},
 54   tmp = this.properties,
 55   self = this,
 56   i = 0,
 57   finalArgs = [],
 58   initArgs = [],
 59   tmpUniques = [],
 60   success = true,
 61   p, doit,
 62   client = self.getClient(),
 63   uniqueLocker = function uniqueLocker(propName, callback) {
 64     if (tmp[propName].unique && tmp[propName].value !== '' && (tmp[propName].__updated || !self.__inDB)) {
 65       var checkCallback = function (err, value) {
 66         if (setDirectly && value) {
 67           tmpUniques.push(Nohm.prefix.unique + self.modelName + ':' + propName + ':' + self.p(propName));
 68         }
 69         if (!setDirectly) {
 70           // client.exists returns 1 if the value exists, client.setnx returns 1 if the value did not exist.
 71           // what we pass to the callback is whether the property has a unique value or if it already exists.
 72           // that means if we used exists we have to use the opposite of the returned value.
 73           value = !value;
 74         }
 75         callback(err, {
 76           p: propName,
 77           unique: value
 78         });
 79       };
 80       
 81       if (setDirectly) {
 82         /*
 83          * We lock the unique value here if it's not locked yet, then later remove the old uniquelock when really saving it. (or we free the unique slot if we're not saving)
 84          */
 85         client.setnx(Nohm.prefix.unique + self.modelName + ':' + propName + ':' + self.p(propName), self.id, checkCallback);
 86       } else {
 87         client.exists(Nohm.prefix.unique + self.modelName + ':' + propName + ':' + self.p(propName), checkCallback);
 88       }
 89     } else {
 90       callback(null, null);
 91     }
 92   };
 93   if (setDirectly && !self.id) {
 94     Nohm.logError('Checking AND setting uniques without having an id set. self:' + require('util').inspect(self));
 95   }
 96   
 97   async.map(Object.keys(tmp), uniqueLocker, function (err, arr) {
 98     if (Array.isArray(arr) && arr.length > 0) {
 99       arr.forEach(function(val) {
100         if (val && ! val.unique) {
101           self.errors[val.p].push('notUnique');
102           success = false;
103         }
104       });
105     }
106     
107     if (setDirectly && ! success) {
108       if (Array.isArray(tmpUniques) && tmpUniques.length > 0) {
109         tmpUniques.forEach(function(val) {
110           // delete tmp unique locks since the new values won't be saved.
111           client.del(val, self.logError);
112         });
113       }
114     }
115     saveCallback(success);
116   });
117 };

139 /**
140  *  Validates a given property.
141  */
142 exports.__validateProperty = function __validateProperty(p) {
143   if (!p || !this.properties[p]) {
144     throw 'Trying to validate undefined property or accessing __validateProperty without giving a property';
145   }
146   if (!this.properties[p].validations) {
147     return true;
148   }
149   var value = this.properties[p].value,
150   validations = this.properties[p].validations,
151   self = this;
152   this.errors[p] = [];
153   
154   return validations.every(function (i) {
155     var valid,
156     errorName = '';
157     if (typeof i === 'function') {
158       var funcName = i.toString().match(/^function ([\w]*)[\s]?\(/);
159       valid = i(value);
160       errorName = 'custom_'+ (funcName[1] ? funcName[1] : p);
161     } else if (typeof i === 'string') {
162       if (!validators[i]) {
163         throw 'Trying to access unavailable validator.';
164       }
165       valid = validators[i](value, []);
166       errorName = i;
167     } else if (i instanceof Array) {
168       if (typeof i[0] === 'function') {
169         valid =  i[0](value, i.slice(1) || []);
170       }
171       if (!validators[i[0]]) {
172         throw 'Trying to access unavailable validator.';
173       }
174       valid = validators[i[0]](value, i.slice(1));
175       errorName = i[0];
176     }
177     if (!valid) {
178       self.errors[p].push(errorName);
179     }
180     return !!valid;
181   });
182 };

190 var validators = {
191   // most of these are copied from the jquery validation plugin http://code.google.com/p/bassistance-plugins/source/browse/trunk/plugins/validate/jquery.validate.js
192 
193   /**
194    * Make sure a value is not empty.
195    */
196   notEmpty: function notEmpty(value) {
197     return (value && value !== true);
198   },
199 
200   /**
201    * Make sure a value is of at least a given length (params[0]) or is optional (params[1]) and empty
202    */
203   minLength: function minLength(value, params) {
204     return (params[1] && !value) || ('' + value).trim().length >= params[0];
205   },
206 
207   /**
208    * Make sure a value is of at most a given length (params[0])
209    */
210   maxLength: function maxLength(value, params) {
211     return ('' + value).trim().length <= params[0];
212   },
213 
214   /**
215    * Make sure a value is of at least a given value (params[0])  (pun intended :P ) or is optional (params[1]) and zero
216    */
217   min: function min(value, params) {
218     return (params[1] && value === 0) || (+value) >= params[0];
219   },
220 
221   /**
222    * Make sure a value is of at most a given value (params[0])
223    */
224   max: function max(value, params) {
225     return (+value) <= params[0];
226   },
227 
228   /**
229    * Make sure a value is a valid email adress or empty if it's optional (params[0])
230    */
231   email: function email(value, params) {
232     return (!value && params[0]) || emailRegexp.test(value);
233   },
234 
235   /**
236    * Make sure a value is a valid url or empty if it's optional (params[0])
237    */
238   url: function url(value, params) {
239     // the fuck is this? (the regex that is now at the start of this file :P )
240     // well, it's copied from jqueries validation plugin and i blindly trust them.
241     return (!value && params[0]) || urlRegexp.test(value);
242   },

244   /**
245    * Make sure a value is a date that the Date object can feed on or is optional (params[0]) and empty
246    */
247   date: function date(value, params) {
248     return (params[0] && !value) || !/Invalid|NaN/.test(new Date(value));
