  7 /**
  8  * This extends an object with x other objects.
  9  * @see http://api.jquery.com/jQuery.extend/
 10  */
 11 Helper.$extend = function() {
 12   var options, name, src, copy, copyIsArray, clone,
 13   target = arguments[0] || {},
 14   i = 1,
 15   length = arguments.length,
 16   deep = false;
 17   
 18   // Handle a deep copy situation
 19   if ( typeof target === "boolean" ) {
 20     deep = target;
 21     target = arguments[1] || {};
 22     // skip the boolean and the target
 23     i = 2;
 24   }
 25   
 26   // Handle case when target is a string or something (possible in deep copy)
 27   if ( typeof target !== "object" && !typeof(target) !== 'function') {
 28     target = {};
 29   }
 30   
 31   for ( ; i < length; i++ ) {
 32     // Only deal with non-null/undefined values
 33     if ( (options = arguments[ i ]) !== null ) {
 34     // Extend the base object
 35       for ( name in options ) {
 36         if (options.hasOwnProperty(name)) { 
 37           src = target[ name ];
 38           copy = options[ name ];
 39   
 40           // Prevent never-ending loop
 41           if ( target === copy ) {
 42             continue;
 43           }
 44     
 45           // Recurse if we're merging plain objects or arrays
 46           if ( deep && copy && ( isPlainObject(copy) || (copyIsArray = Array.isArray(copy)) ) ) {
 47             if ( copyIsArray ) {
 48               copyIsArray = false;
 49               clone = src && Array.isArray(src) ? src : [];
 50     
 51             } else {
 52               clone = src && isPlainObject(src) ? src : {};
 53             }
 54     
 55             // Never move original objects, clone them
 56             target[ name ] = Helper.$extend( deep, clone, copy );
 57     
 58           // Don't bring in undefined values
 59           } else if ( copy !== undefined ) {
 60             target[ name ] = copy;
 61           }
 62         }
 63       }
 64     }
 65   }
 66 
 67 	// Return the modified object
 68 	return target;
 69 };

 71 var isPlainObject = function( obj ) {
 72 
 73 	// Not own constructor property must be Object
 74 	if ( obj.constructor &&
 75 		!obj.hasOwnProperty("constructor") &&
 76 		!obj.constructor.prototype.hasOwnProperty("isPrototypeOf") ) {
 77 		return false;
 78 	}
 79 
 80 	// Own properties are enumerated firstly, so to speed up,
 81 	// if last one is own, then all properties are own.
 82 
 83 	var key;
 84 	for ( key in obj ) {};
 85 
 86 	return key === undefined ||  obj.hasOwnProperty(key);
 87 }

 96 // extend a given object with the given prototypes
 97 Helper.prototypeModule = function (obj, protos) {
 98   for (var proto in protos) {
 99     if (protos.hasOwnProperty(proto)) {
100       obj.prototype[proto] = protos[proto];
101     }
102   }
103   return obj;
104 }

106 /**
107  * Get an object containing all the prefixes for the different key types.
108  */
109 Helper.getPrefix = function (defaultPrefix) {
110   var prefix = null
111   , obj;
112   process.argv.forEach(function (val, index) {
113     if (val.indexOf('test/tests.js') !== -1)
114       prefix = 'tests';
115     if ( ! prefix && val === '--nohm-prefix')
116       prefix = process.argv[index + 1];
117   });
118   if (prefix === null)
119     prefix = defaultPrefix;
120     
121   obj = {
122     ids: prefix + ':ids:',
123     idsets: prefix + ':idsets:',
124     hash: prefix + ':hash:',
125     unique: prefix + ':uniques:',
126     scoredindex: prefix + ':scoredindex:',
127     index: prefix + ':index:',
128     relations: prefix + ':relations:',
129     relationKeys: prefix + ':relationKeys:',
130     meta: prefix + ':meta:'
131   }
132   return obj;
133 }

140 /**
141  * Get the intersection of 2 or more id arrays.
142  * @param {Array[]} Arrays The arrays (containing ids) you want to check
143  */
144 Helper.idIntersection = function idIntersection(first) {
145   var ret = first,
146       empty = false;
147   Array.prototype.slice.call(arguments, 1).forEach(function (arr) {
148     if ( ! Array.isArray(arr) ) {
149       throw new Error('intersections received non-array argument');
150     }
151     if (arr.length === 0) {
152       empty = true;
153     }
154     if (empty || arr === ret) {
155       return false;
156     }
157     
158     ret = arr.filter(function(value) {
159       return ! isNaN(parseInt(value, 10)) && ret.indexOf(value) !== -1;
160     });
161     if (ret.length === 0) {
162       empty = true;
163     }
164   });
165   return empty ? [] : ret;
166 }

170 /**
171  *  Checks if the last item in the given array is a function and returns that or an empty function.
172  */
173 Helper.getCallback = function getCallback(args) {
174   if (args.length >= 1 &&
175       typeof args[args.length - 1] === 'function') {
176     return args[args.length - 1];
177   } else {
178     return function () {};
179   }

182 /**
183  * Checks whether 2 nohm objects are the same.
184  */
185 Helper.checkEqual = function checkEqual(obj1, obj2) {
186   if (!obj1 || (obj1 && !obj2)) {
187     return false;
188   }
189   if (obj1 === obj2) {
190     return true; // hm... is this ever going to trigger? i don't know
191   }
192   else if (obj1.hasOwnProperty('modelName') && obj2.hasOwnProperty('modelName') &&
193            obj1.modelName === obj2.modelName) {
194     // if both have been saved, both must have the same id.
195     if (obj1.id && obj2.id && obj1.id === obj2.id) {
196       return true;
197     }
198     else if (obj1.id && obj2.id) { // both have been saved but do not share the same id -> must be different.
199       return false;
200     }
201 
202     // if both have exactly the same properties (and at most one has been saved - see above)
203     if (obj1.allProperties(true) === obj2.allProperties(true)) {
204       return true;
205     }
206   }
207   return false;
208 };
