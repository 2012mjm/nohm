 10  *  Saves the object by either creating, or updating it.
 11  */
 12 exports.save = function save(callback) {
 13   var self = this,
 14   id_tries = 0,
 15   action = 'create',
 16   idGenerated = false;
 17   var generateId = function () {
 18     idGenerated = true;
 19     var id = parseInt(Math.random() * 100, 10)+''+(parseInt(+ new Date()/1000, 10));
 20     id_tries++;
 21     self.exists(id, function (exists) {
 22       if (exists && id_tries < 500) {
 23         generateId();
 24       } else if ( ! exists) {
 25         _save(id);
 26       } else {
 27         Nohm.logError('Unable to find a new free id after 500 tries.');
 28         callback('no free id found');
 29       }
 30     });
 31   };
 32   var _save = function (id) {
 33     if (id) {
 34       self.id = id;
 35     }
 36     self.valid(false, true, function (valid) {
 37       if (!valid && typeof callback === 'function') {
 38         if (action === 'create') {
 39           self.id = null;
 40         }
 41         callback('invalid');
 42       } else if (valid && action === 'create') {
 43         __create.call(self, callback);
 44       } else if (valid) {
 45         __update.call(self, false, callback);
 46       }
 47     });
 48   };
 49   if (!this.id) {
 50     generateId();
 51   } else {
 52     self.exists(this.id, function (exists) {
 53       if (exists) {
 54         action = 'update';
 55       }
 56       _save();
 57     });
 58   }
 59 };

 66 var __create = function __create(callback) {
 67   var self = this;
 68   this.getClient().incr(Nohm.prefix.ids + this.modelName, function (err, newId) {
 69     if (!err) {
 70       self.getClient().sadd(Nohm.prefix.idsets + self.modelName, newId, function (err) {
 71         if (err) { Nohm.logError(err); }
 72         self.__setUniqueIds(newId, function (err) {
 73           if (err) { Nohm.logError(err); }
 74           self.id = newId;
 75           __update.call(self, true, callback);
 76         });
 77       });
 78     } else {
 79       console.log('Nohm: Creating an object resulted in a client error: ' + util.inspect(err));
 80       if (typeof callback === 'function') {
 81         callback(err);
 82       } else {
 83         throw err;
 84       }
 85     }
 86   });

 89 exports.__index = function __index(p, client) {
 90   var prefix;
 91   client = client || this.getClient();
 92   if (this.properties[p].__numericIndex) {
 93     // we use scored sets for things like "get all users older than 5"
 94     prefix = Nohm.prefix.scoredindex + this.modelName;
 95     if (this.__inDB) {
 96       client.zrem(prefix + ':' + p, this.id, Nohm.logError);
 97     }
 98     client.zadd(prefix + ':' + p, this.properties[p].value, this.id, Nohm.logError);
 99   }
100   prefix = Nohm.prefix.index + this.modelName;
101   if (this.__inDB) {
102     client.srem(prefix + ':' + p + ':' + this.properties[p].__oldValue, this.id, Nohm.logError);
103   }
104   client.sadd(prefix + ':' + p + ':' + this.properties[p].value, this.id, Nohm.logError);

107 /**
108  *  Update an existing dataset with the new values.
109  * @ignore
110  */
111 __update = function __update(all, callback) {
112   var hmsetArgs = [],
113   props = this.properties,
114   self = this,
115   p,
116   multi = this.getClient().multi();
117   
118   hmsetArgs.push(Nohm.prefix.hash + this.modelName + ':' + this.id);
119   for (p in props) {
120     if (all || props[p].__updated) {
121       hmsetArgs.push(p);
122       hmsetArgs.push(props[p].value);
123     }
124   }
125   if (hmsetArgs.length > 1) {
126     multi.hmset.apply(multi, hmsetArgs);
127   }
128   
129   for (p in props) {
130     if (props.hasOwnProperty(p)) {
131       // free old uniques
132       if (props[p].unique === true && props[p].__updated) {
133         if (self.__inDB) {
134           multi.del(Nohm.prefix.unique + self.modelName + ':' + p + ':' + props[p].__oldValue, Nohm.logError);
135         }
136       }
137       if (props[p].index === true && (!self.__inDB || props[p].__updated)) {
138         self.__index(p, multi);
139       }
140     }
141   }
142   multi.exec(function (err) {
143     if (typeof callback !== 'function' && err) {
144       Nohm.logError('Nohm: Updating an object resulted in a client error: ' + err);
145       throw err;
146     } else if (err) {
147       callback(err);
148     } else {
149   
150       // we're using a serial forEach here because otherwise multiple objects 
151       // may error out without notifying the callback
152       async.forEachSeries(self.relationChanges,
153         function (item, cb) {
154           self['__' + item.action](item.object, item.name, function (err, childFail, childName) {
155             item.callback(item.action,
156                           self.modelName,
157                           item.name,
158                           item.object);
159             if (err && childFail) {
160               cb({ err: err, modelName: childName});
161             } else if (err) {
162               cb({ err: err, modelName: item.object.modelName});
163             } else {
164               cb();
165             }
166           });
167         },
168         function (err) {
169           if (typeof callback !== 'function' && err) {
170             Nohm.logError('Nohm: Updating an object resulted in a client error: ' + err);
171             throw err;
172           } else if (err) {
173             callback(err.err, true, err.modelName);
174           } else {
175             callback();
176           }
177         }
178       );
179       
180     }
181   });

184 /**
185  *  Remove an objet from the database.
186  *  Note: Does not destroy the js object or its properties itself!
187  */
188 exports.remove = function remove(callback) {
189   var self = this;
190   
191   if (!this.id) {
192     return callback('The object you are trying to delete has no id.');
193   } else if (!this.__inDB) {
194     this.load(this.id, function (err) {
195       if (err) {
196         return callback(err);
197       } else {
198         return __realDelete.call(self, callback);
199       }
200     });
201   } else {
202     return __realDelete.call(self, callback);
203   }
204 };

206 var __realDelete = function __realDelete(callback) {
207   var self = this;
208   
209   var p, i, len, 
210   multi = self.getClient().multi();
211 
212   multi.del(Nohm.prefix.hash + this.modelName + ':' + this.id);
213   multi.srem(Nohm.prefix.idsets + this.modelName, this.id);
214 
215   for (p in this.properties) {
216     if (this.properties.hasOwnProperty(p)) {
217       if (this.properties[p].unique) {
218         multi.del(Nohm.prefix.unique + this.modelName + ':' + p + ':' +
219                   this.properties[p].__oldValue);
220       }

222         multi.srem(Nohm.prefix.index + self.modelName + ':' + p + ':' +
223                    this.properties[p].__oldValue,

225       }
226       if (self.properties[p].__numericIndex) {
227         multi.zrem(Nohm.prefix.scoredindex + this.modelName + ':' + p,
228                    this.id);
229       }
230     }
231   }
232 
233   this.unlinkAll(multi, function () {
234     multi.exec(function (err, values) {
235       self.id = 0;
