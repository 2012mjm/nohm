  6 /**
  7  *  Get or set a property.
  8  *  This automatically invokes typecasting and behaviours.
  9  *  If you pass true as the third parameter,
 10  *  the property is instantly checked for validity.
 11  */
 12 exports.property = function property(key, val, validate) {
 13   var tmp, old, success = true, p;
 14   if (typeof key === 'object') {
 15     if (typeof(validate) === 'undefined') {
 16       validate = val;
 17     }
 18     for (p in key) {
 19       if (key.hasOwnProperty(p)) {
 20         if (!this.p(p, key[p], validate)) {
 21           success = false;
 22         }
 23       }
 24     }
 25     return success;
 26   }
 27   if (!this.properties[key]) {
 28     Nohm.logError('Trying to access undefined property "'+key+
 29       '" of object "'+this.modelName+'" with id:'+this.id+'.');
 30     return false;
 31   }
 32   tmp = this.properties[key];
 33   if (typeof val === 'undefined') {
 34     return tmp.type === 'json' ? JSON.parse(tmp.value) : tmp.value;
 35   } else if (val !== tmp.value) {
 36     tmp = this.properties[key]
 37     old = tmp.value;
 38     tmp.value = this.__cast(key, val, old);
 39     if (validate) {
 40       if (!this.valid(key)) {
 41         tmp.value = old;
 42         return false;
 43       }
 44     }
 45     tmp.__updated = tmp.value !== tmp.__oldValue;
 46   }
 47   return true;
 48 }

 76 /**
 77    *  Casts a property to a certain datatype. (Might cause unexpected results.
 78    *  Behaviours offer greater control over what happens.)
 79    *  Currently supported:
 80    *  string,
 81    *  integer,
 82    *  float,
 83    *  timestamp (time/date string or number to unix timestamp),
 84    *  json
 85    */
 86 exports.__cast = function __cast(key, value, old) {
 87   if (!this.properties[key]) {
 88     Nohm.logError('Trying to access undefined property "' +
 89     key + '" of object "' + this.modelName + '".');
 90     return false;
 91   }
 92   var type = this.properties[key].type,
 93     timezoneOffset,
 94     matches,
 95     hours,
 96     minutes;
 97 
 98   if (typeof type === 'function') {
 99     return type.call(this, value, key, old);
100   }
101   switch (type) {
102   case 'bool':
103     return value === 'false' ? false : !!value;
104   case 'string':
105     // no .toString() here. TODO: or should there be?
106     return (
107             (!(value instanceof String) ||
108              value.toString() === '') && typeof value !== 'string'
109             ) ? ''
110               : value;
111   case 'integer':
112     return isNaN(parseInt(value, 10)) ? 0 : parseInt(value, 10);
113   case 'float':
114     return isNaN(parseFloat(value)) ? 0 : parseFloat(value);
115   case 'timestamp':
116     // make it a timestamp aka. miliseconds from 1970
117     if (isNaN(value) && typeof value === 'string') {
118       // see if there is a timezone specified in the string
119       matches = value.match(/(\+|\-)([\d]{1,2})\:([\d]{2})$/);
120       if (value.match(/Z$/)) {
121         // UTC timezone in an ISO string (hopefully)
122         timezoneOffset = 0;
123       } else if (matches) {
124         // +/- hours:minutes specified.
125         // calculating offsets in minutes and removing the offset from the string since new Date() can't handle those.
126         hours = parseInt(matches[2], 10);
127         minutes = parseInt(matches[3], 10);
128         if (matches[1] === '-') {
129           timezoneOffset = -1 * (hours * 60 + minutes);
130         } else {
131           timezoneOffset = hours * 60 - minutes;
132         }
133         value = value.substring(0, value.length - matches[0].length);
134       } else {
135         timezoneOffset = new Date(value).getTimezoneOffset();
136       }
137       return new Date(value).getTime() - timezoneOffset * 60 * 1000;
138     }
139     return parseInt(value, 10);
140   case 'json':
141     try {
142       // already is json, no nothing
143       JSON.parse(value);
144       return value;
145     } catch (e) {
146       return JSON.stringify(value);
147     }
148   default:
149     return value;
150   }
151 }

153 /**
154  * Get an array of all properties that have been changed.
155  */
156 exports.propertyDiff = function propertyDiff(key) {
157   var diff = [],
158   p;
159   if (key && !this.properties[key]) {
160     throw 'Invalid key specified for diffProperty';
161   }
162 
163   for (p in this.properties) {
164     if (!key || p === key) {
165       if (this.properties[p].__updated) {
166         diff.push({
167           key: p,
168           before: this.properties[p].__oldValue,
169           after: this.properties[p].value
170         });
171       }
172     }
173   }
174   return diff;
175 }

186     if (!key || p === key) {
